    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // DOM Elements
            const urlInput = document.getElementById('url-input');
            const downloadBtn = document.getElementById('download-btn');
            const optionBtns = document.querySelectorAll('.option-btn');
            const qualityOptions = document.querySelectorAll('.quality-option');
            const previewContainer = document.getElementById('preview-container');
            const previewContent = document.getElementById('preview-content');
            const loading = document.getElementById('loading');
            const errorMessage = document.getElementById('error-message');
            const notificationBar = document.getElementById('notification-bar');
            const notificationMessage = document.getElementById('notification-message');
            const notificationClose = document.getElementById('notification-close');
            
            // Current settings
            let currentType = 'auto';
            let currentQuality = '4k';
            
            // Event Listeners
            downloadBtn.addEventListener('click', processDownload);
            urlInput.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') processDownload();
            });
            
            // Option buttons
            optionBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    optionBtns.forEach(b => b.classList.remove('active'));
                    this.classList.add('active');
                    currentType = this.dataset.type;
                });
            });
            
            // Quality options
            qualityOptions.forEach(option => {
                option.addEventListener('click', function() {
                    qualityOptions.forEach(o => o.classList.remove('active'));
                    this.classList.add('active');
                    currentQuality = this.dataset.quality;
                });
            });
            
            // Download options (preview)
            document.querySelectorAll('.download-option').forEach(option => {
                option.addEventListener('click', function() {
                    // In a real app, this would trigger the actual download
                    alert(Preparing ${currentQuality} ${currentType} download...);
                });
            });
            
            // Notification close button
            notificationClose.addEventListener('click', function() {
                hideNotification();
            });
            
            // Main function to process download
            function processDownload() {
                const url = urlInput.value.trim();
                
                if (!url) {
                    showError('Please enter a valid URL');
                    return;
                }
                
                // Validate URL format
                if (!isValidUrl(url)) {
                    showError('Please enter a valid URL from supported platforms');
                    return;
                }
                
                // Show loading state
                loading.style.display = 'block';
                previewContainer.style.display = 'none';
                errorMessage.style.display = 'none';
                
                // Simulate processing delay
                setTimeout(() => {
                    loading.style.display = 'none';
                    
                    // In a real app, you would fetch and process the content here
                    // This is just a simulation
                    const isVideo = Math.random() > 0.5;
                    const platform = detectPlatform(url);
                    
                    if (Math.random() > 0.1) { // 90% success rate for demo
                        showPreview(isVideo, platform);
                        showNotification(Your ${isVideo ? 'video' : 'photo'} has been enhanced to ${currentQuality.toUpperCase()} and is ready for download!);
                    } else {
                        showError('Content not found or private. Please try another link.');
                    }
                }, 2000);
            }
            
            // Notification functions
            function showNotification(message) {
                notificationMessage.textContent = message;
                notificationBar.classList.add('show');
                
                // Auto-hide after 5 seconds
                setTimeout(() => {
                    hideNotification();
                }, 5000);
            }
            
            function hideNotification() {
                notificationBar.classList.remove('show');
            }
            
            // Helper functions
            function isValidUrl(string) {
                try {
                    new URL(string);
                    return true;
                } catch (_) {
                    return false;
                }
            }
            
            function detectPlatform(url) {
                if (url.includes('youtube.com') || url.includes('youtu.be')) return 'youtube';
                if (url.includes('facebook.com')) return 'facebook';
                if (url.includes('tiktok.com')) return 'tiktok';
                if (url.includes('instagram.com')) return 'instagram';
                if (url.includes('pinterest.com')) return 'pinterest';
                return 'unknown';
            }
            
            function showPreview(isVideo, platform) {
                previewContent.innerHTML = '';
                
                if (isVideo) {
                    // Create video preview
                    const video = document.createElement('video');
                    video.src = getDemoVideoUrl(platform);
                    video.controls = true;
                    video.autoplay = true;
                    video.muted = true;
                    video.style.maxWidth = '100%';
                    previewContent.appendChild(video);
                    
                    // Update download options for video
                    document.querySelector('.download-video span').textContent = 
                        Download Video (${currentQuality.toUpperCase()});
                } else {
                    // Create image preview
                    const img = document.createElement('img');
                    img.src = getDemoImageUrl(platform);
                    img.alt = 'Enhanced preview';
                    img.style.maxWidth = '100%';
                    previewContent.appendChild(img);
                    
                    // Update download options for image
                    let resolution = '32MP';
                    if (currentQuality === 'original') resolution = 'Original';
                    else if (currentQuality === 'hd') resolution = '12MP';
                    else if (currentQuality === '4k') resolution = '24MP';
                    document.querySelector('.download-photo span').textContent = 
                        Download Photo (${resolution});
                }
                
                previewContainer.style.display = 'block';
            }
            
            function showError(message) {
                errorMessage.textContent = message;
                errorMessage.style.display = 'block';
                previewContainer.style.display = 'none';
                loading.style.display = 'none';
            }
            
            // Demo content URLs
            function getDemoVideoUrl(platform) {
                const videos = {
                    youtube: 'https://samplelib.com/lib/preview/mp4/sample-5s.mp4',
                    facebook: 'https://samplelib.com/lib/preview/mp4/sample-10s.mp4',
                    tiktok: 'https://samplelib.com/lib/preview/mp4/sample-15s.mp4',
                    instagram: 'https://samplelib.com/lib/preview/mp4/sample-20s.mp4',
                    pinterest: 'https://samplelib.com/lib/preview/mp4/sample-30s.mp4',
                    unknown: 'https://samplelib.com/lib/preview/mp4/sample-5s.mp4'
                };
                return videos[platform];
            }
            
            function getDemoImageUrl(platform) {
                const images = {
                    youtube: 'https://source.unsplash.com/random/800x450/?youtube',
                    facebook: 'https://source.unsplash.com/random/800x450/?facebook',
                    tiktok: 'https://source.unsplash.com/random/800x450/?tiktok',
                    instagram: 'https://source.unsplash.com/random/800x450/?instagram',
                    pinterest: 'https://source.unsplash.com/random/800x450/?pinterest',
                    unknown: 'https://source.unsplash.com/random/800x450/?social'
                };
                return images[platform];
            }
        });
